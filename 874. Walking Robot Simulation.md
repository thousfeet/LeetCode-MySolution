class Solution {
public:
    int robotSim(vector<int>& commands, vector<vector<int>>& obstacles) {
        int x = 0, y = 0; 
        int towards = 0;
        int move[4][2] = {0,1,1,0,0,-1,-1,0};
        int ans = 0;
        for(auto c : commands)
        {
        	if(c == -2)
        		towards = (towards-1)%4;
			else if(c == -1)
				towards = (towards+1)%4;
			else
			{
				while(c--)
				{
					x += move[towards][0];
					y += move[towards][1];
					for(auto o : obstacles)
					{
						if(x == o[0] && y == o[1])
						{
							x -= move[towards][0];
							y -= move[towards][1];
							break;
						}
					}
					ans = max(ans,x*x+y*y);
				}
			}
		}
        return ans;
    }
};
